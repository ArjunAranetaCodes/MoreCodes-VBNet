//Write a Program that has function which accepts two strings then
//returns the concatenated version. 

Module Module1

    Private Function PrintStrings(ByVal strWord1 As String, ByVal strWord2 As String)
        Dim strNewWord As String = strWord1 & " " & strWord2
        Return strNewWord
    End Function

    Sub Main()
        Console.WriteLine(PrintStrings("Hello", "World"))

        Console.Read()
    End Sub

End Module
